# create manual function for DFT
def dft(x,inverse=False):
    N = len(x)
    n = np.arange(N)
    k = n.reshape((N, 1))
    e_pow = -2j * np.pi * k * n / N
    if inverse:
        return np.dot(np.exp(-e_pow),x)/N
    return np.dot(np.exp(e_pow), x)
TransFunc_dft = dft(transfer_function)
