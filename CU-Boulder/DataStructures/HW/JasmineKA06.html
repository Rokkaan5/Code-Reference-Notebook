def fib_dp(n):
  cheat_sheet = [-1] * (n+1)
  cheat_sheet[0] = 0
  cheat_sheet[1] = 1
  return fib_dp_helper(cheat_sheet,n)

def fib_dp_helper(cheat_sheet,n):
  if cheat_sheet[n] != -1:
    return cheat_sheet[n]
  cheat_sheet[n] = fib_dp_helper(cheat_sheet,n-1) + fib_dp_helper(cheat_sheet,n-2)
  return cheat_sheet[n]
