Using Boolean Arrays

import numpy as np
np.random.seed(1)
ar = np.random.randint(100, size=10)
ar
array([37, 12, 72,  9, 75,  5, 79, 64, 16,  1])
ar_isodd = ar % 2 == 1
ar_isodd
array([ True, False, False,  True,  True,  True,  True, False, False,
        True])
ar[ar_isodd]
array([37,  9, 75,  5, 79,  1])
ar[ar % 2 == 1]
array([37,  9, 75,  5, 79,  1])
